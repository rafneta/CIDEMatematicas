Recta tangente

rec_tan <- function(f,x_0){
  fp <- Deriv(f)
  fpe <-fp(x_0)
  
  rectan <- function(x){
    return(fpe*(x-x_0)+f(x_0))
  }
  
  return(rectan)
}
rt <- rec_tan(f,pi) # Regresa la recta tangente como función
rt(1)
[1] 2.141593

Función para gráfica

library(plotly)
grafica2d <- function(x,y,xp="",yp=""){
  
  datos <- data.frame(x=x, y = y)

  fig <- plot_ly(datos, 
               x = ~x,
               y = ~y, 
               name = "f_t",
               type = 'scatter',
               mode = 'lines') %>%
        layout(title = "Gráfica de la función",
               xaxis =  list(title ="x"))
  
  if(xp[1] != ""){
    datos1 <- data.frame(xp = xp,yp=yp)
    fig <- fig %>% add_trace(data = datos1, 
               x = ~xp,
               y = ~yp, 
               name = "Recta tangente",
               type = 'scatter',
               mode = 'lines',
               opacity = 1) %>%
        layout(title = "Gráfica de la función y recta tangente")
    
  }

  return(fig)
}
x <- seq(from = -10, to = 15, by=0.01 )

Si se revisa la documentación de deriv se observa que están implementadas

D(expr, name)

entre otras cosas se tiene acceso al siguiente código, que es una funci`pn que realiza la derivación de una variable del orden deseado

DD <- function(expr, name, order = 1) {
   if(order < 1) stop("'order' must be >= 1")
   if(order == 1) D(expr, name)
   else DD(D(expr, name), name, order - 1)
}
DD(expression(sin(x^2)), "x", 3)
-(sin(x^2) * (2 * x) * 2 + ((cos(x^2) * (2 * x) * (2 * x) + sin(x^2) * 
    2) * (2 * x) + sin(x^2) * (2 * x) * 2))
deriv3(y ~ b0 + b1 * 2^(-x/th), c("b0", "b1", "th"))
expression({
    .expr3 <- 2^(-x/th)
    .expr6 <- log(2)
    .expr7 <- th^2
    .expr9 <- .expr6 * (x/.expr7)
    .expr10 <- .expr3 * .expr9
    .value <- b0 + b1 * .expr3
    .grad <- array(0, c(length(.value), 3L), list(NULL, c("b0", 
        "b1", "th")))
    .hessian <- array(0, c(length(.value), 3L, 3L), list(NULL, 
        c("b0", "b1", "th"), c("b0", "b1", "th")))
    .grad[, "b0"] <- 1
    .grad[, "b1"] <- .expr3
    .hessian[, "b1", "b1"] <- 0
    .hessian[, "b1", "th"] <- .hessian[, "th", "b1"] <- .expr10
    .grad[, "th"] <- b1 * .expr10
    .hessian[, "th", "th"] <- b1 * (.expr10 * .expr9 - .expr3 * 
        (.expr6 * (x * (2 * th)/.expr7^2)))
    attr(.value, "gradient") <- .grad
    attr(.value, "hessian") <- .hessian
    .value
})
